import streamlit as st import pandas as pd import numpy as np import matplotlib.pyplot as plt import koreanize_matplotlib from pathlib import Path from sklearn.ensemble import RandomForestClassifier from sklearn.model_selection import train_test_split # Apply Korean font to matplotlib koreanize_matplotlib.use() # Suppress warnings for cleaner output in Streamlit st.set_option('deprecation.showPyplotGlobalUse', False) # --- Re-define functions from the original notebook --- # # These functions are necessary for the Streamlit app to run independently def load_and_preprocess_data(data_path): """ 데이터를 로드하고 전처리합니다. - timestamp 컬럼을 datetime으로 변환합니다. - measure_deviation과 spec_margin 컬럼을 계산합니다. """ df = pd.read_csv(data_path) df["timestamp"] = pd.to_datetime(df["timestamp"]) df["measure_deviation"] = df["measure_value"] - (df["lsl"] + df["usl"]) / 2 df["spec_margin"] = np.minimum(df["measure_value"] - df["lsl"], df["usl"] - df["measure_value"]) return df def calculate_cpk(df): """ 주어진 DataFrame에서 품목별 Cpk를 계산합니다. """ def calc_cpk_single_group(group): mean = group["measure_value"].mean() std = group["measure_value"].std() usl = group["usl"].iloc[0] lsl = group["lsl"].iloc[0] if std == 0: return 0.0 cpu = (usl - mean) / (3 * std) cpl = (mean - lsl) / (3 * std) return round(min(cpu, cpl), 3) cpk_df = df.groupby("product_type").apply(calc_cpk_single_group).reset_index(name="cpk").sort_values("cpk") return cpk_df def generate_control_chart_data(df, target_product): """ 특정 품목의 관리도 시각화에 필요한 데이터를 준비합니다. 측정값, 평균, UCL, LCL을 포함하는 DataFrame을 반환합니다. """ chart_df = df.loc[df["product_type"] == target_product].sort_values("timestamp").copy() mean_value = chart_df["measure_value"].mean() std_value = chart_df["measure_value"].std() ucl = mean_value + 3 * std_value lcl = mean_value - 3 * std_value control_chart_data = pd.DataFrame({ "timestamp": chart_df["timestamp"], "measure_value": chart_df["measure_value"], "mean_value": mean_value, "ucl": ucl, "lcl": lcl }) return control_chart_data def summarize_warning_and_process_vars(df): """ 품목별 경고 비율과 주요 공정 변수(온도, 압력, 습도)의 평균을 요약합니다. """ warning_summary = ( df.groupby("product_type", as_index=False) .agg( warning_ratio=("warning_flag", "mean"), avg_temp=("temperature_c", "mean"), avg_pressure=("pressure_bar", "mean"), avg_humidity=("humidity_pct", "mean"), ) ) warning_summary["warning_ratio"] = (warning_summary["warning_ratio"] * 100).round(2) return warning_summary def train_early_warning_model(df, model_cols, target_col): """ 조기 경보 모델(RandomForestClassifier)을 학습시키고 학습된 모델 객체를 반환합니다. """ X = df[model_cols] y = df[target_col] X_train, X_test, y_train, y_test = train_test_split( X, y, test_size=0.25, random_state=42, stratify=y ) warning_model = RandomForestClassifier( random_state=42, n_estimators=150, max_depth=5, min_samples_leaf=3, ) warning_model.fit(X_train, y_train) return warning_model, X_test, y_test def predict_top_risk_lots(model, df, model_cols, n_top_lots=10): """ 학습된 모델을 사용하여 각 lot의 경고 확률을 예측하고, 경고 확률이 높은 상위 N개 lot을 반환합니다. """ risk_table = df.copy() risk_table["warning_probability"] = model.predict_proba(df[model_cols])[:, 1] top_risk_lots = ( risk_table.sort_values("warning_probability", ascending=False) .loc[:, ["timestamp", "product_type", "lot_id", "measure_value", "warning_flag", "warning_probability"]] .head(n_top_lots) ) return top_risk_lots def predict_new_samples(model, new_samples_df, model_cols): """ 새로운 샘플에 대해 경고 예측을 수행하고 확률을 계산합니다. """ samples_copy = new_samples_df.copy() samples_copy["pred_warning"] = model.predict(samples_copy[model_cols]) samples_copy["warning_probability"] = model.predict_proba(samples_copy[model_cols])[:, 1] return samples_copy # --- Streamlit App --- # st.title("🏭 제조 품질 SPC 대시보드") # Sidebar for file upload st.sidebar.header("데이터 업로드") uploaded_file = st.sidebar.file_uploader("CSV 파일을 업로드하세요", type=["csv"]) if uploaded_file is not None: # Use uploaded file spc_df = load_and_preprocess_data(uploaded_file) st.sidebar.success("파일이 성공적으로 업로드되었습니다.") else: # Fallback to default data path candidate_paths = [ Path("/content/SPC 통계분석/quality_process_spc.csv"), Path("../data/quality/quality_process_spc.csv"), Path("../../data/quality/quality_process_spc.csv"), ] data_path = None for path in candidate_paths: if path.exists(): data_path = path break if data_path is None: st.error("기본 'quality_process_spc.csv' 파일을 찾을 수 없습니다. 파일을 업로드하거나 경로를 확인하세요.") st.stop() else: spc_df = load_and_preprocess_data(data_path) st.sidebar.info("업로드된 파일이 없어 기본 데이터를 사용합니다.") # Display raw data st.subheader("원본 데이터 미리보기") st.write(f"데이터 크기: {spc_df.shape[0]} 행, {spc_df.shape[1]} 열") st.dataframe(spc_df.head()) # Cpk Calculation and Display st.subheader("품목별 Cpk 분석") cpk_results = calculate_cpk(spc_df) st.dataframe(cpk_results.style.apply(lambda x: ['background-color: #ffe6e6' if v < 1.0 else ('background-color: #fff2cc' if v < 1.33 else '') for v in x], subset=['cpk'])) st.markdown("--- :orange[**Cpk 1.0 미만**] :red[**위험**] | :orange[**Cpk 1.0 ~ 1.33 미만**] :yellow[**주의**] | :green[**Cpk 1.33 이상**] :blue[**안정**] --- ") # Cpk Bar Chart st.subheader("품목별 Cpk 비교") fig, ax = plt.subplots(figsize=(8, 4)) ax.bar(cpk_results["product_type"], cpk_results["cpk"], color=plt.cm.Paired(np.arange(len(cpk_results)))) # Use a colormap for varied colors ax.axhline(1.33, color="green", linestyle="--", linewidth=1, label="1.33 기준") ax.axhline(1.00, color="black", linestyle="--", linewidth=1, label="1.00 기준") ax.set_title("품목별 Cpk 비교") ax.set_ylabel("Cpk") ax.legend() plt.tight_layout() st.pyplot(fig)
